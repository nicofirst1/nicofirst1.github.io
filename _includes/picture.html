{% comment %} Responsive picture element with WebP fallback. External URLs, SVGs, and WebPs are served as plain . Usage: {% include picture.html src=item.image alt=item.title class="my-class" loading="lazy" %} {% endcomment %} {% assign img_ext = include.src | split: '.' | last %} {% assign is_external = false %} {% if include.src contains 'http://' or include.src contains 'https://' %}{% assign is_external = true %}{% endif %} {% if is_external %} {% elsif img_ext == 'svg' or img_ext == 'webp' %} {% else %} {% assign webp_src = include.src | replace: img_ext, 'webp' | relative_url %} {% endif %}
